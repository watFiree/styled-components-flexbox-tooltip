### Write

```jsx
import * as flex from "styled-components-flexbox-tooltip";

const Wrapper = styled.div`
  ${flex.CenterCenterColumn};
  background-color: red;
  padding: 1.1rem;
`;
```

### Instead of

```jsx
import * as flex from "styled-components-flexbox-tooltip";

const Wrapper = styled.div`
  display: flex;
  align-items: center;
  justify-content: center;
  flex-directon: column;
  background-color: red;
  padding: 1.1rem;
`;
```
